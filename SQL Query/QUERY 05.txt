QUERY 5
-- Business Problem: Assess how service usage (Internet + Security) impacts customer churn.
SELECT
    InternetService,
    OnlineSecurity,
    COUNT(*) as customer_count,
    SUM(CASE WHEN Churn THEN 1 ELSE 0 END) as churned,
    ROUND(SUM(CASE WHEN Churn THEN 1.0 ELSE 0 END) / COUNT(*) * 100, 2) as churn_rate
FROM `elaborate-art-471212-d9.Telco_Customer_Churn.Cleaned_Dataset`
GROUP BY InternetService, OnlineSecurity
HAVING COUNT(*) >= 50
ORDER BY churn_rate DESC;
